@PREFIX=rtccount
@NADDR=1
@SLAVE.TYPE=SINGLE
@SLAVE.BUS=wb
@$STEP=(1<<32)/@$(CLOCK.FREQUENCY)
@MAIN.DEFNS=
	reg	[31:0]	r_@$(PREFIX)_data;
@MAIN.INSERT=
	initial	r_@$(PREFIX)_data = 32'h0;
	always @(posedge i_clk)
		r_@$(PREFIX)_data <= r_@$(PREFIX) + @$(STEP);
	assign	@$(SLAVE.PREFIX)_idata = r_@$(PREFIX)_data;
@REGS.N=1
@REGS.0= 0 R_RTCCOUNT RTCCOUNT
